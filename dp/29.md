class Solution {
public:
    int dfs(int x, string & s, int n) {
        int res = 0;

        if(x >= n) return 1;
        int x1 = s[x] - '0';
        if(x1 == 0) return 0;
        else res += dfs(x+1, s, n);

        if(x + 1 >= n) return 1;
        int x2 = (s[x] - '0') * 10 + s[x+1] - '0';
        if(s[x] == '0') return 0; 
        else if(x2 > 26) res += 0; // 这里本来是return 0
        // 根据样例

int mem[102];
class Solution {
public:
    int dfs(int x, string & s, int n) {
        if(mem[x] != -1) return mem[x];
        int res = 0;

        if(x >= n) return 1;
        int x1 = s[x] - '0';
        if(x1 == 0) return 0;
        else res += dfs(x+1, s, n);

        if(x + 1 >= n) return 1;
        int x2 = (s[x] - '0') * 10 + s[x+1] - '0';
        if(s[x] == '0') return 0;
        else if(x2 > 26) res += 0;
        else res += dfs(x+2, s, n);

        mem[x] = res;
        return res;
    }
    int numDecodings(string s) {
        int n = s.size();
        memset(mem, -1, sizeof mem);
        return dfs(0, s, n);

    }
};
